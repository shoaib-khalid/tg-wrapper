    Introduction

    This documentation aims to provide all the information you need to work with our API.

    Base URL

    https://tgw.symplified.biz

    Authenticating requests

    This API is not authenticated.

    Endpoints

    Receive a newly incoming request from telegram client

    Example request:
    curl --request POST \
    "https://tgw.symplified.biz/telegram/incoming" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"callback_query\": {
        \"message\": {
            \"chat\": {
                \"id\": 7
            },
            \"from\": {
                \"username\": \"error\"
            },
            \"data\": \"consequatur\"
        }
    },
    \"message\": {
        \"chat\": {
            \"id\": 10
        },
        \"from\": {
            \"username\": \"voluptatum\"
        },
        \"data\": \"in\"
    }
}"

    const url = new URL(
    "https://tgw.symplified.biz/telegram/incoming"
);

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "callback_query": {
        "message": {
            "chat": {
                "id": 7
            },
            "from": {
                "username": "error"
            },
            "data": "consequatur"
        }
    },
    "message": {
        "chat": {
            "id": 10
        },
        "from": {
            "username": "voluptatum"
        },
        "data": "in"
    }
}

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
